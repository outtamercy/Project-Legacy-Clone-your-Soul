Scriptname BennyPhylacteryScript extends ObjectReference

ObjectReference Property aaBennySoulBindingStandRef Auto
Quest Property aaBennyAlternateDeathQuest Auto
GlobalVariable Property aaBennySoulBound Auto
Message Property aaBennyPhylacteryMessage Auto
Message Property aaBennySoulBoundMessage Auto
Message Property aaBennyUnbindSoulMessage Auto
Message Property aaBennySoulUnboundMessage Auto
ActorBase Property Player Auto

Event OnActivate(ObjectReference akActionRef)
    If !aaBennyAlternateDeathQuest.GetStageDone(5)
        int bindSoul = aaBennyPhylacteryMessage.Show()
        If bindSoul == 0
            aaBennySoulBound.SetValueInt(1)
            Player.SetEssential()
            aaBennyAlternateDeathQuest.SetStage(5)
            aaBennySoulBindingStandRef.PlayAnimation("Open")
            aaBennySoulBoundMessage.Show()
        EndIf
    ElseIf aaBennyAlternateDeathQuest.GetStageDone(5)
        aaBennySoulBindingStandRef.PlayAnimation("StartClose")
        Utility.wait(1.0)
        aaBennySoulBindingStandRef.PlayAnimation("PickUp")
        int unbindSoul = aaBennyUnbindSoulMessage.Show()
        Utility.wait(0.5)
        aaBennySoulBindingStandRef.PlayAnimation("SetDown")
        If unbindSoul == 0
            aaBennySoulBound.SetValueInt(0)
            Player.SetEssential(False)
            aaBennyAlternateDeathQuest.SetStage(10)
            aaBennyAlternateDeathQuest.Reset()
            aaBennySoulUnboundMessage.Show()
        Else
            Utility.wait(0.5)
            aaBennySoulBindingStandRef.PlayAnimation("Open")
        EndIf
    EndIf
EndEvent
